126 void OMMotor::ms( uint8_t p_Div ) {
127

129  bool s2 = false;
130  bool s3 = false;
131 
132  uint8_t wasMs = m_curMs;
133  m_curMs = p_Div;
134 
135  switch( p_Div ) {
136  case 2:
137  s1 = true;
138  break;
139  case 4:
140  s2 = true;
141  break;
142  case 8:
143  s1 = true;
144  s2 = true;
145  break;
146  case 16:
147  s1 = true;

149  s3 = true;

1100 void OMMotor::move(bool p_Dir, unsigned long p_Steps) {
1101 
1102 
1103  // if motor is disabled, do nothing
1104  if( ! enable() || ( maxSteps() > 0 && stepsMoved() >= maxSteps() ) ) {
1105  _fireCallback(OM_MOT_DONE);
1106  return;
1107  }
1108 
1109  // note: the check on p_Steps is required
1110  // to allow manual moves that are not continuous
1111  // when motor is set in continuous mode
1112 
1113  if( p_Steps == 0 && continuous() ) {
1114  // continuous motion mode
1115  if( ! running() ) {
1116  m_asyncSteps = 0;
1117  _stepsAsync(p_Dir, (unsigned long) 0);
1118  }
1119 
1120  _fireCallback(OM_MOT_BEGIN);
1121  return;
1122  }
1123 
1124  // continuous allows higher stepping rate
1125  // so clamp back down if not doing a continuous
1126  // move
1127 
1128  if( maxStepRate() > 5000 )
1129  maxStepRate(5000);
1130 
1131  // check for backlash compensation
1132  if( m_backCheck == true ) {
1133  p_Steps += backlash();
1134  m_backCheck = false;
1135  }
1136 
1137  // not in continuous motion mode
1138  // nothing to do
1139  if( p_Steps == 0 ) {
1140  _fireCallback(OM_MOT_DONE);
1141  return;
1142  }
1143 
1144  if( ! m_calcMove ) {
1145 
1146  // this move was not specified with a time to accomplish, so accomplish
1147  // in minimal amount of time possible
1148 
1149  // determine how far we'd move in our acceleration and deceleration
1150  // times for moves that must operate at maximum speed. We accelerate
1151  // and decel over 200 steps each, or 1/4 of the total steps, whichever is less
1152 
1153  // set travel const (.travel) here before attempting to use it
1154 
1155  _setTravelConst(&m_splineOne);
1156 
1157  unsigned int mSpeed = ( maxStepRate() > maxSpeed() ) ? maxSpeed() : maxStepRate();
1158 
1159  float rampSteps = (200 > (p_Steps / 4)) ? (p_Steps / 4) : 200;
1160 
1161  rampSteps *= 2;
1162 
1163  float crTm = ((p_Steps - rampSteps) / mSpeed) * 1000;
1164  float adTm = ((rampSteps / mSpeed) * 1000) * m_splineOne.travel;
1165 
1166  float mvMS = (crTm + adTm) + 1.0;
1167 
1168  // take a minimum of 50ms to make the move - to prevent over-speeding
1169  // and getting goofy.
1170  if( mvMS < 50.0 ) {
1171  mvMS = 50.0;
1172  }
1173 
1174 
1175  // one spline point per ms, like normal.
1176  m_totalSplines = (unsigned long) mvMS;
1177 
1178  // prep spline variables
1179  _initSpline(false, p_Steps, mvMS, adTm/2, adTm/2);
1180 
1181  // we need to initialize the first spline point
1182  m_curSpline = 1;
1183  float tmPos = (float) m_curSpline / (float) m_totalSplines;
1184  // calculate the first running speed
1185  f_easeFunc(false, tmPos);
1186  }
1187 
1188  // limit step cycle using this control
1189  // value, without mucking with max steps
1190 
1191  m_asyncSteps = p_Steps;
1192 
1193  // re-set called from five-argument flag
1194 
1195  m_calcMove = false;
1196 
1197  _stepsAsync(p_Dir, p_Steps);
1198  _fireCallback(OM_MOT_MOVING);
1199 
1200 }

1343 
1344 void OMMotor::_runISR() {
1345 
1346 
1347  // digital pin 9
1348  static float cycleErrAccumulated = 0.0;
1349  static unsigned long cyclesLow = 0;
1350  static unsigned long stepsTaken = 0;
1351  static unsigned long totalCyclesTaken = 0;
1352 
1353  if( m_refresh ) {
1354  cyclesLow = 0;
1355  stepsTaken = 0;
1356  totalCyclesTaken = 0;
1357  cycleErrAccumulated = 0.0;
1358  m_refresh = false;
1359  }
1360 
1361  if( m_asyncSteps > 0 && totalCyclesTaken >= m_cyclesPerSpline ) {
1362 
1363  // we are ready for the next point in the spline,
1364  // run speed calculations
1365 
1366  if( m_asyncSteps > 0 && m_curSpline == m_totalSplines ) {
1367  // hey, look at that - we're at the end of our spline (and
1368  // we haven't finished our last step either, otherwise we
1369  // wouldn't get here...)
1370 
1371  if( stepsTaken < m_asyncSteps ) {
1372  // we really should be taking any steps we're missing. (if asked to
1373  // move a specific distance.) positioning errors can add up over
1374  // multiple moves, and we shouldn't leave expected steps on the
1375  // cutting-room floor even if we have exceeded the destination time
1376 
1377  // TODO: multiple final steps will happen at the fastest step rate,
1378  // whether or not the motor can handle this! We really shouldn't be
1379  // more than one step shy of our target, however, unless one specifies
1380  // a move that requires a step rate higher than the maximum step rate,
1381  // which is just painful.
1382 
1383  // directly access register to bring pin high
1384  OM_MOT_STPREG |= (1 << OM_MOT_STPFLAG);
1385 
1386  totalCyclesTaken--;
1387  stepsTaken++;
1388  _updateMotorHome( (uint8_t) 1);
1389  // bring step pin low - this allows us to
1390  // hit one step per ISR run rate. (Timing is important
1391  // we hit minimum high pulse time in the instructions above
1392  // this line)
1393  OM_MOT_STPREG &= (B11111111 ^ (1 << OM_MOT_STPFLAG));
1394  return;
1395  }
1396 
1397  // we've reached the end of our anticipated time. Go ahead and finish
1398  // everything up, and stop the timer. (we should only get here when a
1399  // last step had to be taken above.)
1400 
1401  stepsTaken = 0;
1402  m_curSpline = 0;
1403  totalCyclesTaken = 0;
1404  cycleErrAccumulated = 0.0;
1405  cyclesLow = 0;
1406 
1407  stop();
1408  return;
1409  } // end if( m_curSpline...
1410 
1411  // move to the next point in the current spline.
1412  m_curSpline++;
1413  float tmPos = (float) m_curSpline / (float) m_totalSplines;
1414 
1415  // get new off cycle timing for the next point in the spline
1416  f_easeFunc(false, tmPos);
1417  totalCyclesTaken = 0;
1418 
1419 
1420 
1421  // we don't stop here, as we still need to check the off-time
1422  // between steps, based on our current spline point
1423 
1424  } // end if( m_asyncSteps > 0 && totalCyclesTaken...
1425 
1426 
1427  totalCyclesTaken++;
1428 
1429  if( cycleErrAccumulated >= 1.0 ) {
1430 
1431 
1432  // check the error rate (fractions of a cycle that
1433  // were accumulated - if at least one full cycle, add
1434  // an additional delay cycle to get overall movement
1435  // timing as close to exact as is possible
1436 
1437  cycleErrAccumulated -= 1.0;
1438  // run an extra cycle low
1439  return;
1440  }
1441 
1442  // increase lowcycle counter after applying error correction, as we
1443  // don't want to take the next step too fast
1444 
1445  cyclesLow++;
1446 
1447  if( cyclesLow >= m_curOffCycles ) {
1448  // we've had enough low cycles, ok to trigger next step
1449 
1450  // set pin to trigger step
1451  // digital pin 9 on atmega328p
1452  OM_MOT_STPREG |= (1 << OM_MOT_STPFLAG);
1453 
1454  cyclesLow = 0;
1455  stepsTaken++;
1456 
1457  _updateMotorHome(1);
1458 
1459  // accumulate cycle off time errors
1460  // once per complete cycle (fractions of
1461  // off cycles are included in the calculated
1462  // speed)
1463 
1464  cycleErrAccumulated += m_curCycleErr;
1465 
1466  // if we hit the step count requested for this move,
1467  // or if we have hit the maximum stepping point,
1468  // stop now - don't overshoot
1469  if( (m_totalSteps > 0 && m_stepsMoved >= m_totalSteps) || (m_asyncSteps > 0 && stepsTaken >= m_asyncSteps) ) {
1470  stepsTaken = 0;
1471  cycleErrAccumulated = 0.0;
1472  cyclesLow = 0;
1473  stop();
1474  }
1475 
1476  // bring step pin low again, the driver will register
1477  // a pulse of only a few uS, so we can perform
1478  // a complete step in a single timing cycle,
1479  // allowing us to hit a speed of one step per
1480  // timing cycle
1481 
1482  OM_MOT_STPREG &= (B11111111 ^ (1 << OM_MOT_STPFLAG));
