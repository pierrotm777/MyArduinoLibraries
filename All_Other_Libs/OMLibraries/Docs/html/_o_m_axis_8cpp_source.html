603 bool OMAxis::move(bool p_dir, unsigned long p_steps, unsigned long p_travel, unsigned long p_accel, unsigned long p_decel) {
604 
605  OMByteBuffer sendBuf;
606  uint8_t _buf[21];
607 
608  sendBuf.assign(_buf, sizeof(_buf));
609 
610  sendBuf.append(CMD_PC_MOVE_COMPLEX);
611  sendBuf.append(p_dir);
612  sendBuf.append(p_steps);
613  sendBuf.append(OM_PDU_ESC);
614  sendBuf.append(p_travel);
615  sendBuf.append(OM_PDU_ESC);
616  sendBuf.append(p_accel);
617  sendBuf.append(OM_PDU_ESC);
618  sendBuf.append(p_decel);
619 
620  return ( command(m_slaveAddr, OM_PCODE_PC, sendBuf.getBuf(), sendBuf.size()) > 0 );
621 
622 
623 }
