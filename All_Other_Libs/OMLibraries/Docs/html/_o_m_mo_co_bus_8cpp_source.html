265 uint8_t OMMoCoBus::getPacket() {
266 
267  uint8_t command_val = 0;
268  uint8_t com_byte_count = 0;
269  //uint8_t dat = 0;
270  uint8_t ret = 0;
271 
272  m_isBCast = false;
273 
274  bool not_us = false;
275 
276  // we check serial.available() here even
277  // though this->_getNextByte() does, because
278  // a timeout is applied waiting for next command
279  // in this->_getNextByte() - since this is called
280  // every cycle of the main loop, we don't want
281  // to introduce a delay unless we know of a
282  // byte waiting to be read
283 
284  int avail = m_serObj->available();
285 
286  // serial does not have data in the buffer
287  if( avail <= 0 )
288  return(0);
289 
290  // is this packet intended for us?
291  uint8_t stat = this->_targetUs();
292 
293  if(stat == OM_SER_ERR || stat == OM_SER_TIMEOUT) {
294  // got an error? timeout error, really.
295  // this->_flushSerial();
296  return(0);
297  }
298  else if( stat == OM_SER_IS_BCAST ) {
299  // command was a broadcast command
300  m_isBCast = true;
301  }
302  else if( stat != OM_SER_OK ) {
303  // command was for someone else
304  // (but we need to read the rest of the data
305  // off the line to make sure we don't keep part of it
306  // in the buffer)
307  not_us = true;
308  }
309 
310  // get command byte
311  ret = this->_getNextByte(command_val);
312 
313  if ( ret != OM_SER_OK ) {
314  // well, this is just embarassing
315  // this->_flushSerial();
316  return(0);
317  }
318 
319  if( command_val > 0 ) {
320  // get data length (even if not our own command)
321  ret = this->_getNextByte(com_byte_count);
322 
323  if ( ret != OM_SER_OK ) {
324  // well, this is just embarassing
325  // this->_flushSerial();
326  return(0);
327  }
328  }
329 
330  // check for overflow
331  // note: we will still be in error condition, as
332  // there will be values left in the serial buffer
333  // on the next read, from this command - so don't
334  // exceed it, eh? This is just to prevent memory
335  // corruption. It is not safe otherwise
336 
337  if(com_byte_count > OM_SER_BUFLEN)
338  com_byte_count = OM_SER_BUFLEN;
339 
340 
341  // clear out any previous command data
342  memset(m_serBuffer, 0, sizeof(uint8_t) * OM_SER_BUFLEN);
343  m_bufSize = 0;
344 
345 
346  // populate command data buffer
347  for( int i = 1; i <= com_byte_count; i++ ) {
348  // get com_byte_count character values from the serial
349  // buffer
350 
351  uint8_t dat = 0;
352  uint8_t ret = this->_getNextByte(dat);
353 
354  if( ret != OM_SER_OK )
355  return(0);
356 
357  m_serBuffer[i - 1] = dat;
358  m_bufSize++;
359  }
360 
361  if( not_us == true ) {
362  // the command was not intended for us
363  memset(m_serBuffer, 0, sizeof(uint8_t) * OM_SER_BUFLEN);
364  return(0);
365  }
366 
367 
368  return( command_val );
369 }

600 uint8_t OMMoCoBus::_targetUs() {
601 
602  uint8_t thsChar = 1;
603  uint8_t ret = this->_getNextByte(thsChar);
604 
605  // timeout? other error?
606  if( ret != OM_SER_OK )
607  return(OM_SER_TIMEOUT);
608 
609  // if character is null...
610  if( thsChar == 0 ) {
611  // until we get enough nulls for a start command
612  // or read something that isn't null...
613 
614  uint8_t nullCount = 0;
615 
616  while( nullCount < 5 && thsChar == 0 ) {
617  nullCount++;
618 
619  // null count met
620  if( nullCount == 5 )
621  break;
622 
623  ret = this->_getNextByte(thsChar);
624 
625  if( ret != OM_SER_OK )
626  return(OM_SER_TIMEOUT);
627 
628  }
629 
630  // did not receive a proper start sequence
631  if( nullCount < 5 )
632  return(OM_SER_ERR);
633 
634  thsChar = 0;
635 
636  ret = this->_getNextByte(thsChar);
637 
638  if( ret != OM_SER_OK )
639  return(OM_SER_TIMEOUT);
640 
641  if( thsChar != 255 )
642  return(OM_SER_ERR);
643 
644  // get target address
645 
646  unsigned int addr = 0;
647  ret = this->_getNextByte(thsChar);
648 
649  if( ret != OM_SER_OK )
650  return(OM_SER_TIMEOUT);
651 
652  addr = (unsigned int) thsChar << 8;
653 
654  ret = this->_getNextByte(thsChar);
655 
656  if( ret != OM_SER_OK )
657  return(OM_SER_TIMEOUT);
658 
659  addr |= (unsigned int) thsChar;
660 
661  // is not destined for us?
662 
663  if( addr == OM_SER_BCAST_ADDR )
664  return(OM_SER_IS_BCAST);
665  else if( addr != m_devAddr )
666  return(OM_SER_NOT_US);
667 
668  } // end if(thsChar == 0
669  else {
670  // uint8_t invalid
671  return(OM_SER_ERR);
672  }
673 
674  return(OM_SER_OK);
675 
676 }

678 uint8_t OMMoCoBus::_getNextByte(uint8_t& p_data) {
679 
680  uint8_t ret = OM_SER_OK;
681  uint8_t* p = (uint8_t*)(void*) &p_data;
682 
683  if( m_serObj->available() > 0 ) {
684  *p = (uint8_t) m_serObj->read();
685  }
686  else {
687  unsigned long timer = millis();
688  // wait until we have data ready
689  while( m_serObj->available() < 1 ) {
690  if( millis() - timer > OM_SER_WAIT ) {
691  // timed out waiting for complete input
692  return( OM_SER_TIMEOUT );
693  }
694  }
695  *p = (uint8_t) m_serObj->read();
696  }
697 
698 
699  return(ret);
700 }
